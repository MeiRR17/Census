# הגדרת אופן פעולה אוטומטי ללא שאלות אישור (מונע תקיעות של הסקריפט)
option batch abort
option confirm off

# פתיחת חיבור מאובטח לשרת הסטנדאלון מבוסס לינוקס
# פורמט: sftp://username:password@ip_address/
open sftp://root:YOUR_PASSWORD_HERE@192.168.x.x/

# יצירת תיקיית עבודה בשרת (אם היא לא קיימת) ומעבר אליה
# אנו בוחרים בנתיב מסודר תחת /opt כמקובל בסביבות שרתים
mkdir /opt/axlerate_gateway
cd /opt/axlerate_gateway

# הגדרת התיקייה המקומית במחשב שלך שממנה ניקח את הקבצים
lcd .\export_build_axlerate

# העלאת קובץ ה-TAR במצב בינארי. זה קריטי כדי שהקובץ לא ייפגם בהעברה
put -transfer=binary axlerate-image.tar

# העלאת קבצי התצורה
put docker-compose.yml
put .env

# --- ביצוע פקודות מרחוק על שרת הסטנדאלון ---
# הפקודה call מאפשרת לנו להריץ פקודות טרמינל (CLI) על השרת ישירות מתוך WinSCP
call echo "Loading Docker image into Standalone Server..."
# טעינת התמונה מתוך קובץ ה-TAR אל תוך מנוע הדוקר של השרת
call docker load -i axlerate-image.tar

# עצירת שירות ישן (במידה וקיים כזה מאתמול או מהבדיקה הקודמת)
call docker-compose down

# העלאת השירות החדש באוויר במצב מנותק (Detached mode)
call docker-compose up -d

# סגירת החיבור ויציאה נקייה
close
exit
